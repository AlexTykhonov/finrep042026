// 1. Налаштування бази const db = new Dexie("AuditDB"); db.version(2).stores({ reports: "++id, edrpou, companyName, period" }); const parser = new XMLParser({ ignoreAttributes: false }); // 2. Логіка парсингу саме вашого формату async function processFiles(files) { const progressBar = document.getElementById('progressBar'); const statusText = document.getElementById('statusText'); document.getElementById('progressArea').style.display = 'block'; for (let file of files) { addLog(`Читаю архів: ${file.name}`); const arrayBuffer = await file.arrayBuffer(); const zip = fflate.unzipSync(new Uint8Array(arrayBuffer)); let batch = []; for (let path in zip) { if (path.toLowerCase().endsWith('.xml')) { const xmlText = new TextDecoder().decode(zip[path]); const obj = parser.parse(xmlText); // Шлях до даних з вашого скріншоту const head = obj.DECLAR?.DECLARHEAD; const body = obj.DECLAR?.DECLARBODY; if (body) { const edrpou = body.FIRM_EDRPOU || head?.TIN || "00000000"; const name = body.FIRM_NAME || "Невідома назва"; const year = head?.PERIOD_YEAR || "---"; // Витягуємо ключові фінансові показники для "зірочок" // Наприклад: 1495 (Активи), 2000 (Виручка) const assets = parseFloat(body.A1495) || 0; const revenue = parseFloat(body.A2000) || 0; batch.push({ edrpou: String(edrpou).trim(), companyName: String(name).trim(), period: String(year), assets: assets, revenue: revenue, raw: body // зберігаємо весь об'єкт для детального перегляду }); } } if (batch.length > 200) { await db.reports.bulkPut(batch); batch = []; } } if (batch.length > 0) await db.reports.bulkPut(batch); addLog(`Архів ${file.name} оброблено.`); } statusText.innerText = "Готово! Можна шукати."; } // 3. Функція пошуку з вашим дизайном (Картки + Зірочки) async function searchCompany() { const query = document.getElementById('searchInput').value.trim(); const resultsArea = document.getElementById('resultsArea'); resultsArea.innerHTML = "Шукаю..."; const results = await db.reports.where('edrpou').equals(query).toArray(); if (results.length === 0) { resultsArea.innerHTML = "❌ Компанію з таким ЄДРПОУ не знайдено."; return; } resultsArea.innerHTML = results.map(r => { // Визначаємо найбільший показник для зірочки const starAssets = r.assets > 1000000 ? "⭐ (Великі активи)" : ""; return `

${r.companyName}

Рік: ${r.period} | ЄДРПОУ: ${r.edrpou}

Усього активів (ряд. 1495):
${r.assets.toLocaleString()} ${starAssets}
Дохід (ряд. 2000):
${r.revenue.toLocaleString()}
`; }).join(''); }
